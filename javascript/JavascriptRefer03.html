        document.write("함수가 실행되었습니다.");

    function callback(str){

            str();
        }
    }
    callback(func);
}

19. 콜백 함수 : 동기/비동기

{
    //01
    document.write("*** 19. 콜백 함수 : 동기/비동기 ***");
    
    function funA(){
        console.log("funA가 실행되었습니다.");
    }
    function funB(){
        console.log("funB가 실행되었습니다.");
    }
    funB();
    funA();

    //02
    // function funC(){
    //     setTimeout(function(){
    //         console.log("funC가 실행되었습니다.");
    //     },1000);
    // }
    function funC(){
        setTimeout(()= > {
            console.log("funC가 실행되었습니다.");
        },1000);
    }
    function funD(){
        console.log("funD가 실행되었습니다.");
    }
    funC();
    funD();

    //03
    function funE(callback){
        setTimeout(() = > {
            console.log("funE가 실행되었습니다.");
            callback();
        }, 1000);
    }
    function funF(){
        console.log("funF가 실행되었습니다.");
    }
    funE(funF);
}
